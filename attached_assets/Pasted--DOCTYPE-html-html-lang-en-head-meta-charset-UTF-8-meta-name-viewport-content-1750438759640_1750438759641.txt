    <script>
        let allSelected = false;

        function toggleCheckbox(checkbox) {
            checkbox.classList.toggle('checked');
            updateGuestCount();
        }

        function toggleSelectAll() {
            allSelected = !allSelected;
            const checkboxes = document.querySelectorAll('.guest-checkbox');
            const button = document.querySelector('.select-all-btn');
            
            checkboxes.forEach(checkbox => {
                if (allSelected) {
                    checkbox.classList.add('checked');
                } else {
                    checkbox.classList.remove('checked');
                }
            });
            
            button.textContent = allSelected ? 'Deselect All' : 'Select All';
            updateGuestCount();
        }

        function updateGuestCount() {
            const totalGuests = document.querySelectorAll('.guest-item').length;
            const selectedGuests = document.querySelectorAll('.guest-checkbox.checked').length;
            const countElement = document.getElementById('guestCount');
            
            if (selectedGuests > 0) {
                countElement.textContent = `${selectedGuests} of ${totalGuests} selected`;
                countElement.style.background = '#f57c00';
            } else {
                countElement.textContent = `${totalGuests} guests`;
                countElement.style.background = '#667eea';
            }
        }

        // Search functionality
        document.getElementById('searchInput').addEventListener('input', function(e) {
            const searchTerm = e.target.value.toLowerCase();
            const guestItems = document.querySelectorAll('.guest-item');
            
            guestItems.forEach(item => {
                const guestName = item.querySelector('.guest-name').textContent.toLowerCase();
                const guestDetails = item.querySelector('.guest-details').textContent.toLowerCase();
                
                if (guestName.includes(searchTerm) || guestDetails.includes(searchTerm)) {
                    item.style.display = 'flex';
                } else {
                    item.style.display = 'none';
                }
            });
        });

        // Filter functionality
        document.getElementById('statusFilter').addEventListener('change', function(e) {
            const filterValue = e.target.value;
            const guestItems = document.querySelectorAll('.guest-item');
            
            guestItems.forEach(item => {
                const status = item.querySelector('.guest-status').textContent.toLowerCase();
                
                if (filterValue === 'all' || status.includes(filterValue)) {
                    item.style.display = 'flex';
                } else {
                    item.style.display = 'none';
                }
            });
        });

        document.getElementById('commentFilter').addEventListener('change', function(e) {
            const filterValue = e.target.value;
            const guestItems = document.querySelectorAll('.guest-item');
            
            guestItems.forEach(item => {
                const details = item.querySelector('.guest-details').textContent.toLowerCase();
                const hasComments = !details.includes('no comments');
                
                if (filterValue === 'all' || 
                    (filterValue === 'with' && hasComments) || 
                    (filterValue === 'without' && !hasComments)) {
                    item.style.display = 'flex';
                } else {
                    item.style.display = 'none';
                }
            });
        });
    </script>
